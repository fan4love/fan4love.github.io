deepl翻译

功能描述：通过deepl的api实现英译中，将英文与中文拼接，重新写入新的txt文件

#01.实现deepl翻译，封装功能函数
#deepl_txt.py

import requests
from fastlid import fastlid


def Translator_deepl(text):

    lang = fastlid(text)
    if lang == "zh":
        from_lang, to_lang = "zh", "en"
    elif lang == "en":
        from_lang, to_lang = "en", "zh"
    else:
        from_lang = lang[0]
        to_lang = "zh"

    data = [text, from_lang, to_lang]
    json_data = {"data": data}

    url = "https://hf.space/embed/mikeee/gradio-deepl/+/api/predict"
    resp = requests.post(url, json=json_data)
    trans = resp.json()


    return (f"译:{trans['data'][0]}")

#02.实现中英文的拼接，将结果写入空白文件：note01.txt
#translator_deepl_txt.py

from deepl_txt import Translator_deepl

#打开需要翻译的txt文件：newspaper02.txt

with open("newspaper02", "r",encoding="utf-8") as f:

                            english = f.read()
            
with open("note01", "w",encoding="utf-8") as f2:
        while len(english) > 0:

                eng = english.split('.')[0]

                 f2.write(str(eng+"\n"))

                 chinese=Translator_deepl(eng)

                 f2.write(str(chinese+"\n"))
                 english = english[len(english.split('.')[0]) + 1:]

        f2.close()
